def Appointment.in_time_slot_only(params)
  params.inject(Appointment) { |booking_scope, (key, value)|
    case key.to_sym
      when :start
        booking_scope.where("start_time >= ?", value.to_date)
      when :end
        booking_scope.where("end_time   <= ?", value.to_date)
      else
        booking_scope
    end
  }
end
